# Home Web Tour Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a first-visit onboarding tour on the home page that auto-runs once, never auto-runs again after completion/skip, and can be manually replayed.

**Architecture:** Keep onboarding local to `Index.tsx` by mounting a focused `HomeTour` component powered by `react-joyride`. Use a small storage utility for versioned completion state and safe fallback behavior when browser storage is unavailable. Mark tour targets with `data-tour` attributes to keep selectors stable.

**Tech Stack:** React 18, TypeScript, React Router 6, Vitest, Testing Library, react-joyride

```ts
export const HOME_ONBOARDING_KEY = "onboarding.home.v1.completed";

export function readOnboardingCompleted(key: string): boolean {
  try {
    return window.localStorage.getItem(key) === "1";
  } catch {
    return false;
  }
}

export function writeOnboardingCompleted(key: string, completed: boolean): void {
  try {
    window.localStorage.setItem(key, completed ? "1" : "0");
  } catch {
    // intentionally no-op; onboarding must never block app usage
  }
}
```

- [ ] **Step 4: Implement `HomeTour` component minimally**

Create `src/components/onboarding/HomeTour.tsx`:

```tsx
import { useEffect, useMemo, useState } from "react";
import Joyride, { CallBackProps, STATUS } from "react-joyride";
import { HOME_ONBOARDING_KEY, readOnboardingCompleted, writeOnboardingCompleted } from "@/lib/onboardingStorage";

const HomeTour = () => {
  const [hasMounted, setHasMounted] = useState(false);
  const [isRunning, setIsRunning] = useState(false);

  useEffect(() => {
    setHasMounted(true);
    const completed = readOnboardingCompleted(HOME_ONBOARDING_KEY);
    setIsRunning(!completed);
  }, []);

  const steps = useMemo(
    () => [
      { target: '[data-tour="home-hero"]', content: "Chao mung ban den voi trang chu." },
      { target: '[data-tour="home-search"]', content: "Day la khu vuc tim kiem nhanh." },
      { target: '[data-tour="home-featured-tours"]', content: "Kham pha cac tour noi bat." },
      { target: '[data-tour="home-featured-dishes"]', content: "Mon an dac trung theo vung." },
      { target: '[data-tour="home-cta"]', content: "Bat dau hanh trinh cua ban tai day." },
    ],
    []
  );

  const onJoyrideCallback = (data: CallBackProps) => {
    if (data.status === STATUS.FINISHED || data.status === STATUS.SKIPPED) {
      writeOnboardingCompleted(HOME_ONBOARDING_KEY, true);
      setIsRunning(false);
    }
  };

  if (!hasMounted) return null;

  return (
    <>
      <Joyride
        run={isRunning}
        steps={steps}
        continuous
        showProgress
        showSkipButton
        disableScrolling={false}
        callback={onJoyrideCallback}
        locale={{ back: "Quay lai", close: "Dong", last: "Hoan tat", next: "Tiep", skip: "Bo qua" }}
      />

      <button
        type="button"
        className="fixed bottom-4 right-4 z-40 rounded-full bg-primary px-4 py-2 text-sm font-medium text-primary-foreground shadow-lg"
        onClick={() => setIsRunning(true)}
      >
        Xem huong dan
      </button>
    </>
  );
};

export default HomeTour;
```
